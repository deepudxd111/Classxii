Q:Write a program to count number of lines which begins with letter A in a text file story.txt

def count_lines():
    f = open('story.txt', 'r')
    D=f.readlines()
    count = 0
    for line in D:
        if line.startswith('A') or line.startswith('a'):
            count += 1
    f.close()
    print("Number of lines which begins with A",count)

count_lines()
