Q:Write a program to create a binary file student.dat with functions add student and display all record.


import pickle
def student_data():
        roll=int(input("enter roll:"))
        name = input("Enter name:")
        marks = int(input("Enter marks:"))
        data = [roll, name, marks]
        f=open("stud.dat", "ab")
        pickle.dump(data, f)
        f.close()

def display_data():
    f=open("stud.dat", "rb")
    try:
        while True:
            stud = pickle.load(f)
            print(stud)

    except EOFError:
        pass
    f.close()

while True:
    ch=int(input("Enter 1.add student\n2.display all student\n3.exit\n"))
    if ch==1:
        student_data()
    elif ch==2:
        display_data()
    elif ch==3:
        break
    else:
        print("Invalid choice")
